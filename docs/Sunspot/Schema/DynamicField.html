def method_missing(name, *args, &block)
  if name.to_s =~ /\?$/ && args.empty?
    if @field_variants.any? { |variant| "#{variant.attribute}?" == name.to_s }
      'true'
    else
      'false'
    end
  else
    super(name.to_sym, *args, &block)
  end
end

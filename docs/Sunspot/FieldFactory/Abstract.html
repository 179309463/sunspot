def initialize(name, options = {}, &block)
  @name = name.to_sym
  @data_extractor =
    if block
      DataExtractor::BlockExtractor.new(&block)
    else
      DataExtractor::AttributeExtractor.new(options.delete(:using) || name)
    end
end
